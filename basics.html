CREATE

new

create table new_table_name (
    column1 datatype,
    column2 datatype,
    column3 datatype,
   ....
);

Example data types:

  • varchar(255)
  • number(38,0)
  • int

from existing

create table new_table_name as 
    select column1, column2,...
    from existing_table_name
    where .... ;

INSERT

SELECT

SELECT column_name(s)
FROM table_name
WHERE condition
GROUP BY column_name(s)
HAVING condition
ORDER BY column_name(s);

case when

CASE
    WHEN condition1 THEN result1
    WHEN condition2 THEN result2
    WHEN conditionN THEN resultN
    ELSE result
END;

aggregate functions

SELECT 
  AVG(col) as avg, 
  MAX(col) as max, 
  SUM(col) as sum
FROM table;
SELECT COUNT(col) * 100.0 / COUNT(*) AS asdf
FROM table;

group by

SELECT col, count(*)
FROM table
GROUP BY col;

order by

SELECT col1
FROM table
ORDER BY col2 DESC;

having

The HAVING clause was added to SQL because the WHERE keyword could not be used with aggregate functions.

SELECT col1
FROM table
GROUP BY col1
HAVING COUNT(col2) > 10;

where

where and or

SELECT col1
FROM table
WHERE (col2 = 1 OR col2 = 2)
AND (col3 = 'A' OR col3 = 'B');

where between

SELECT col1
FROM table
WHERE col2 BETWEEN 1 AND 2; 
--BETWEEN is inclusive, i.e. the beginning and end values are included

where in

SELECT col1
FROM table
WHERE col2 IN (1, 10, 100);

where is

-- can also use IS NOT
SELECT col1
FROM table
WHERE col2 IS NULL;

where like

-- can also use NOT LIKE
SELECT col1
FROM table
WHERE col2 LIKE 'asdf%';

where like any

select *
from table
where col like any (
  'this %',
  'that %'
)
